\begin{spell}
\spellcarac{name}{Confusion}
\spellcarac{level}{4}
\spellcarac{duree}{12 rounds}
\spellcarac{portee}{36 m}
\spellcarac{description}{Ce sort rend confuses 3d6 cibles dans un rayon de 18 m, incapables par
elles-mêmes de décider de leur action.

\begin{itemize}
\item
  \textbf{Cibles de 2+1 DV ou plus :} Elles peuvent effectuer un jet de
  sauvegarde contre les sorts chaque round pour résister à l'effet du
  sort, agissant librement ce round-ci en cas de réussite.
\item
  \textbf{Cibles de 2 DV ou moins :} Elles ne peuvent pas effectuer de
  jet de sauvegarde.
\item
  \textbf{Comportement :} Effectuez à chaque round un jet sur le tableau
  suivant pour chaque cible afin de voir ce qu'elle fait ce round-ci :
\end{itemize}
\centering
\begin{tabular}{ll}
\titlecell \textbf{2d6} & \titlecell \textbf{Comportement} \\
2--5 & Attaque le groupe du lanceur \\
6--8 & Aucune action \\
9--12 & Attaque le groupe de la cible \\
\end{tabular}
}
\end{spell}
