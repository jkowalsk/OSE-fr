\begin{spell}
\spellcarac{name}{Image miroir}
\spellcarac{level}{2}
\spellcarac{duree}{6 tours}
\spellcarac{portee}{Le lanceur}
\spellcarac{description}{1d4 copies illusoires du lanceur apparaissent.

\begin{itemize}
\item
  \textbf{Comportement :} Les images miroirs ressemblent en tout point
  au lanceur et se comportent comme lui.
\item
  \textbf{Attaques contre le lanceur :} Détruit une image illusoire, que
  l'attaque soit réussie ou non.
\end{itemize}
}
\end{spell}
