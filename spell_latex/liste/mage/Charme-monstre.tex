\begin{spell}
\spellcarac{name}{Charme-monstre}
\spellcarac{level}{4}
\spellcarac{duree}{Voir description}
\spellcarac{portee}{36 m}
\spellcarac{description}{Le sort place un enchantement sur 3d6 créatures de 3 DV ou moins, ou sur
une unique créature de plus de 3 DV. La créature doit faire un jet de
sauvegarde contre les sorts ou être charmé comme suit :

\begin{itemize}
\item
  \textbf{Amitié :} La cible considère le lanceur comme un allié et un
  ami digne de confiance, qu'elle défendra si nécessaire.
\item
  \textbf{Ordres :} S'ils ont une langue en commun, le lanceur peut
  donner des ordres à la créature charmée, qui obéira.
\item
  \textbf{Nature de la cible :} La créature peut s'abstenir d'accomplir
  des actes contraires à sa nature ou à son alignement.
\item
  \textbf{Ordres suicidaires :} Les ordres suicidaires ou à l'évidence
  trop dangereux ne sont jamais suivis.
\end{itemize}

\textbf{Restrictions :} Morts-vivants sont immunisés.

\textbf{Durée :} Le charme dure indéfiniment, mais la cible effectue un jet 
de sauvegarde contre les sorts à intervalles réguliers, en fonction de
son score d'INT. Lorsque l'un de ces jets réussit, le sort prend fin.

\begin{itemize}
\item
  \textbf{INT 3--8 :} La créature effectue un jet de sauvegarde chaque
  mois.
\item
  \textbf{INT 9--12 :} La créature effectue un jet de sauvegarde chaque
  semaine.
\item
  \textbf{INT 13--18 :} La créature effectue un jet de sauvegarde chaque
  jour.
\end{itemize}
}
\end{spell}
