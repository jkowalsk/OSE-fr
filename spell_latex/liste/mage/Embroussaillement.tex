\begin{spell}
\spellcarac{name}{Embroussaillement}
\spellcarac{level}{4}
\spellcarac{duree}{Permanente}
\spellcarac{portee}{36 m}
\spellcarac{description}{Ce sort fait pousser une robuste végétation naturelle dans une zone à
portée. Les plantes se trouvant dans la zone affectée grossissent et
envahissent la zone, devenant une jungle épineuse entremêlée.

\begin{itemize}
\item
  \textbf{Zone :} Une zone d'une taille maximale de 270 m² -- choisie
  par le lanceur -- est affectée (par exemple, 45 m x 6 m, 30 m x 9 m,
  etc.).
\item
  \textbf{Passage :} Seules les très grandes créatures peuvent forcer le
  passage.
\end{itemize}
}
\end{spell}
