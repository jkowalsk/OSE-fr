\begin{spell}
\spellcarac{name}{Bulle anti-magie}
\spellcarac{level}{6}
\spellcarac{duree}{12 tours}
\spellcarac{portee}{Le lanceur}
\spellcarac{description}{Une barrière anti-magie est créée autour du lanceur.

\begin{itemize}
\item
  \textbf{Bloque la magie :} Aucun sort ou effet de sort ne peut passer
  au travers de cette barrière, que ce soit depuis l'intérieur ou depuis
  l'extérieur.
\item
  \textbf{Annulation :} Le lanceur peut mettre fin à ce sort à tout
  moment avant la fin normale du sort.
\end{itemize}
}
\end{spell}
