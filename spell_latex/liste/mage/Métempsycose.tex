\begin{spell}
\spellcarac{name}{Métempsycose}
\spellcarac{level}{5}
\spellcarac{duree}{Spécial}
\spellcarac{portee}{Le lanceur}
\spellcarac{description}{Le corps du lanceur entre dans une transe comateuse tandis que sa force
vitale est transférée dans un réceptacle (tout objet inanimé à moins de
9 m). 

De là, le lanceur de sorts peut tenter de posséder les corps
d'autres créatures.

\begin{itemize}
	\item \textbf{Si le lanceur se trouve dans le réceptacle:}
\begin{itemize}
\item
  \textbf{Possession :} Le lanceur peut tenter de posséder le corps
  d'une autre créature dans un rayon de 36 m. La victime peut effectuer
  un jet de sauvegarde contre les sorts pour résister à la possession.
\item
  \textbf{Retour :} À n'importe quel moment, le lanceur peut choisir de
  retourner dans son propre corps, mettant ainsi fin au sort.
\item
  \textbf{Si le réceptacle est détruit :} Le lanceur meurt.
\item
  \textbf{Si le corps du lanceur est détruit :} Sa force vitale est
  bloquée dans le réceptacle.
\end{itemize}
\item \textbf{Si le lanceur possède une victime :}
\begin{itemize}
\item
  \textbf{Contrôle :} Le lanceur acquiert un contrôle total sur le corps
  de la victime, mais cette dernière n'est pas pour autant en mesure de
  lancer des sorts.
\item
  \textbf{Si le réceptacle est détruit :} Le sort prend fin et la force
  vitale du lanceur reste piégée dans le corps possédé.
\item
  \textbf{Si la victime possédée est tuée :} La force vitale du lanceur
  retourne dans le réceptacle.
\item
  \textbf{Dissipation :} Dissiper le mal oblige la force vitale du
  lanceur à retourner dans le réceptacle.
\item
  \textbf{Si le corps du lanceur est détruit :} Sa force vitale est
  bloquée dans le corps de la créature possédée.
\end{itemize}
\end{itemize}
}
\end{spell}
