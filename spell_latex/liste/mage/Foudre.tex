\begin{spell}
\spellcarac{name}{Foudre}
\spellcarac{level}{3}
\spellcarac{duree}{Instantanée}
\spellcarac{portee}{54 m}
\spellcarac{description}{Un puissant éclair d'énergie électrique, de 18 m de long et de 1,50 m de
large, est invoqué.

\begin{itemize}
\item
  \textbf{Dégâts :} Les créatures touchées par la foudre subissent 1d6
  points de dégâts par niveau du lanceur, un jet de sauvegarde contre
  les sorts réussi divisant les dégâts par deux.
\item
  \textbf{Rebond :} Si la foudre frappe une barrière solide avant
  d'avoir atteint sa pleine longueur, elle est réfléchie et parcourt le
  reste de sa longueur en direction du lanceur.
\end{itemize}
}
\end{spell}
