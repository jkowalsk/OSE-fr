\begin{spell}
\spellcarac{name}{Porte dimensionnelle}
\spellcarac{level}{4}
\spellcarac{duree}{1 round}
\spellcarac{portee}{3 m}
\spellcarac{description}{
Le lanceur ou une unique créature est instantanément transféré à un
endroit situé à une distance maximale de 108 m. La destination peut être
choisie de deux manières :
\begin{enumerate}
\item
  \textbf{Endroit connu :} Un endroit à portée connu du lanceur.
\item
  \textbf{Endroit inconnu :} Spécifié par une série d'indications (par
  exemple, 36 m nord, 48 m est, 24 m plus haut), la somme ne pouvant
  excéder 108 m.
\end{enumerate}

\textbf{Conditions :} Il faut prendre en compte :
\begin{itemize}
\item
  \textbf{Si la destination est occupée :} Par un corps solide, le sort
  échoue.
\item
  \textbf{Si la cible est non consentante :} Elle peut effectuer un jet
  de sauvegarde contre les sorts afin de résister à la téléportation.
\end{itemize}
}
\end{spell}
