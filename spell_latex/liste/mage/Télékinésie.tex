\begin{spell}
\spellcarac{name}{Télékinésie}
\spellcarac{level}{5}
\spellcarac{duree}{6 rounds}
\spellcarac{portee}{36 m}
\spellcarac{description}{
En se concentrant, le lanceur est capable de déplacer des objets ou des
créatures par le seul pouvoir de la pensée.

\begin{itemize}
\item
  \textbf{Poids :} Un poids allant jusqu'à 200 pièces par niveau de
  lanceur peut être ciblé.
\item
  \textbf{Déplacement :} La cible peut être déplacée jusqu'à 6 m par
  round, dans la direction souhaitée par le lanceur (y compris
  verticalement).
\item
  \textbf{Sauvegarde :} Si une créature est ciblée, elle peut effectuer
  un jet de sauvegarde contre les sorts afin de résister au sort.
\item
  \textbf{Concentration :} Si la concentration du lanceur est brisée, la
  cible retombe.
\end{itemize}
}
\end{spell}
