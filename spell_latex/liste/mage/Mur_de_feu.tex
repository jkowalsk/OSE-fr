\begin{spell}
\spellcarac{name}{Mur de feu}
\spellcarac{level}{4}
\spellcarac{duree}{Concentration}
\spellcarac{portee}{18 m}
\spellcarac{description}{Un rideau immobile et opaque d'un feu violet chatoyant jaillit à portée.

\begin{itemize}
\item
  \textbf{Taille :} Le mur peut être de n'importe quelle taille, et de
  toute forme et dimension choisies par le lanceur, jusqu'à 108 m². Par
  exemple, il peut être droit ou recourbé en un cercle protecteur.
\item
  \textbf{Localisation :} Le mur ne peut être invoqué sur des objets
  présents.
\item
  \textbf{Monstres de moins de 4 DV :} Ils sont incapables de traverser
  le mur de feu.
\item
  \textbf{Monstres de 4 DV ou plus :} Ils peuvent traverser le mur, mais
  subissent 1d6 points de dégâts. Le mur inflige le double de dégâts aux
  morts-vivants et aux créatures utilisant le froid ou y étant
  accoutumées.
\item
  \textbf{Concentration :} Le mur perdure tant que le lanceur reste
  concentré et immobile.
\end{itemize}
}
\end{spell}
