\begin{spell}
\spellcarac{name}{Lecture de la magie}
\spellcarac{level}{1}
\spellcarac{duree}{1 tour}
\spellcarac{portee}{Le lanceur}
\spellcarac{description}{Grâce à lecture de la magie, le lanceur peut déchiffrer des runes et
inscriptions magiques comme suit :

\begin{itemize}
\item
  \textbf{Parchemin :} Permet de comprendre l'inscription magique d'un
  parchemin contenant un sort arcanique. Le lanceur peut activer le
  parchemin quand il le souhaite à l'avenir.
\item
  \textbf{Livre de sorts :} Permet de déchiffrer le livre de sorts d'un
  autre lanceur de sorts arcaniques.
\item
  \textbf{Inscriptions magiques :} Permet de comprendre les runes et
  mots magiques inscrits sur un objet ou une surface.
\item
  \textbf{Lectures multiples :} Une fois une inscription lue au moyen de
  lecture de la magie, le lanceur est ensuite capable de la déchiffrer
  sans avoir à nouveau recours au sort.
\end{itemize}
}
\end{spell}
