\begin{spell}
\spellcarac{name}{Dissipation de la magie}
\spellcarac{level}{3}
\spellcarac{duree}{Instantanée}
\spellcarac{portee}{36 m}
\spellcarac{description}{Dissipation de la magie met fin aux sorts non-instantanés présents dans
une zone de 6 m3.

\begin{itemize}
\item
  \textbf{Niveau du lanceur :} Si le niveau du lanceur dont le sort doit
  être dissipé est supérieur au niveau du lanceur de dissipation de la
  magie, il y a 5 \% de chances par différence de niveau que la
  tentative de dissipation échoue.
\item
  \textbf{Objets magiques :} Ils ne sont pas affectés.
\end{itemize}
}
\end{spell}
