\begin{spell}
\spellcarac{name}{Nuage létal}
\spellcarac{level}{5}
\spellcarac{duree}{6 tours}
\spellcarac{portee}{9 m}
\spellcarac{description}{Un brouillard toxique jaillit du bout des doigts du lanceur.

\begin{itemize}
\item
  \textbf{Zone :} Le brouillard remplit une zone de 9 m de diamètre.
\item
  \textbf{Déplacement :} Le brouillard se déplace de 18 m par tour (6 m
  par round), entraîné par le vent (ou loin du lanceur de sorts,
  lorsqu'il n'y a pas de vent).
\item
  \textbf{Tombée :} Le brouillard étant plus lourd que l'air, il
  s'enfonce vers le niveau le plus bas du terrain, remplissant même les
  tanières ou les avens.
\item
  \textbf{Dégâts :} Toutes les créatures en contact avec les vapeurs
  subissent 1 point de dégâts par round de contact.
\item
  \textbf{Créatures de moins de 5 DV :} Elles doivent également
  effectuer un jet de sauvegarde contre la mort (une fois par round de
  contact) ou mourir.
\end{itemize}
}
\end{spell}
