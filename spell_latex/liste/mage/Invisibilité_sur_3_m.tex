\begin{spell}
\spellcarac{name}{Invisibilité sur 3 m}
\spellcarac{level}{3}
\spellcarac{duree}{Permanente}
\spellcarac{portee}{36 m}
\spellcarac{description}{Une créature choisie et toutes les créatures à une distance de 3 m
d'elle deviennent invisibles :

\begin{itemize}
\item
  \textbf{Zone :} La zone d'effet du sort de 3 m de rayon se déplace
  avec la créature choisie.
\item
  \textbf{Quitter la zone :} Les créatures qui se déplacent à plus de 3
  m de la créature choisie redeviennent visibles.
\item
  \textbf{Entrée dans la zone :} Les créatures qui entrent dans la zone
  d'effet après le lancer du sort ne deviennent pas invisibles.
\item
  \textbf{Équipement :} Tout équipement porté par la cible est également
  rendu invisible. Les objets lâchés ou déposés par une créature
  invisible redeviennent visibles. Les sources de lumière transportées
  sont invisibles, mais pas la lumière qu'elles projettent.
\item
  \textbf{Briser l'invisibilité :} Si une cible affectée attaque ou
  lance un sort, son invisibilité est brisée.
\end{itemize}
}
\end{spell}
