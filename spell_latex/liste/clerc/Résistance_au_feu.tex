\begin{spell}
\spellcarac{name}{Résistance au feu}
\spellcarac{level}{2}
\spellcarac{duree}{2 tours}
\spellcarac{portee}{9 m}
\spellcarac{description}{Ce sort accorde à une créature une résistance surnaturelle au feu, comme
suit :

\begin{itemize}
\item
  \textbf{Chaleur normale :} Les créatures ne sont pas affectées par la
  chaleur ou le feu non magiques.
\item
  \textbf{Bonus de sauvegarde :} La créature reçoit un bonus de +2 à
  tous ses jets de sauvegarde contre les attaques magiques ou les
  souffles basés sur le feu.
\item
  \textbf{Dégâts de feu :} Ils sont réduits de 1 point par dé de dégâts
  lancé (chaque dé inflige un minimum de 1 point de dégâts).
\end{itemize}
}
\end{spell}
