\begin{spell}
\spellcarac{name}{Pierre en boue}
\spellcarac{level}{5}
\spellcarac{duree}{3d6 jours}
\spellcarac{portee}{36 m}
\spellcarac{description}{
\subsubsection{Transformation de pierre en boue}
Ce sort transforme un volume de jusqu'à 270 m$^2$ de pierre en boue de 3 m
de profondeur.

\textbf{Déplacement :} Vitesse réduite de 90\%.


\subsubsection{Inversé : Transformation de boue en pierre}

Transforme un volume identique de boue en pierre. Cette altération est
permanente.
}
\end{spell}
