\begin{spell}
\spellcarac{name}{Porte dimensionnelle}
\spellcarac{level}{4}
\spellcarac{duree}{1 round}
\spellcarac{portee}{3 m}
\spellcarac{description}{
Le lanceur ou une unique créature est instantanément transféré à un
endroit situé à une distance maximale de 108 m. La destination peut être :
\begin{enumerate}
\item
  \textbf{Connue :} Un endroit à portée connu du lanceur.
\item
  \textbf{Inconnue :} Spécifiée par une série d'indications
  (p.e. 36 m nord, 48 m est, 24 m plus haut), la somme ne pouvant excéder 108 m.
\end{enumerate}

Si la \textbf{destination est occupée} par un corps solide, le sort
  échoue.\\
Si la \textbf{Cible est non consentante} un jet desauvegarde contre les sorts annule le sort.

}
\end{spell}
