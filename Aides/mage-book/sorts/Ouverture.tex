\begin{spell}
\spellcarac{name}{Ouverture}
\spellcarac{level}{2}
\spellcarac{duree}{1 round}
\spellcarac{portee}{18 m}
\spellcarac{description}{Le sort ouverture permet d'ouvrir les portes coincées, barrées ou
verrouillées, les portails, les coffres, etc.~

\begin{itemize}
\item
  \textbf{Portes fermées magiquement :} Elles sont affectées par ce sort
  (ex. fermeture, verrou magique).
\item
  \textbf{Portes secrètes :} Le sort peut les ouvrir à condition que le
  lanceur soit conscient de leur existence.
\end{itemize}
}
\end{spell}
