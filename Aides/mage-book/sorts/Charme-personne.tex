\begin{spell}
\spellcarac{name}{Charme-personne}
\spellcarac{level}{1}
\spellcarac{duree}{1+ jours}
\spellcarac{portee}{36 m}
\spellcarac{smalldesc}{Voit lanceur comme ami; save sorts annule}
\spellcarac{description}{Une créature humanoïde doit réussir
un jet de sauvegarde contre les sorts ou se retrouver charmée :

\begin{itemize}
\item
  \textbf{Amitié :} La cible considère le lanceur comme un
  ami digne de confiance, qu'elle défendra si nécessaire.
\item
  \textbf{Ordres :} S'ils ont une langue en commun, la créature charmée obéira aux
  ordres du lanceur.
\item
  \textbf{Nature de la cible :} La créature n'accomplit pas
  d'acte contraire à sa nature ou à son alignement.
\item
  \textbf{Ordres suicidaires :} Les ordres suicidaires ou à l'évidence
  trop dangereux ne sont jamais suivis.
\end{itemize}

\textbf{Restrictions :} Les monstres humanoïdes de plus de 4+1 DV et les
morts-vivants ne sont pas affectés.

\textbf{Durée :} Infinie, mais la cible effectue un jet
de sauvegarde contre les sorts à intervalles réguliers, en fonction de
son score d'INT. Si un jet réussit, le sort prend fin.

\begin{itemize}
  \item \textbf{INT 3-8 :} Chaque mois.
  \item \textbf{INT 9-12 :} Chaque semaine.
  \item \textbf{INT 13-18 :} Chaque jour.
\end{itemize}
}
\end{spell}
