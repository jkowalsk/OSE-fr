\begin{spell}
\spellcarac{name}{Contrôle du climat}
\spellcarac{level}{6}
\spellcarac{duree}{Concentration}
\spellcarac{portee}{240 m}
\spellcarac{description}{En se concentrant (aucun mouvement n'est autorisé), le lanceur modifie
le climat autour de lui pour provoquer l'un des effets suivants.
Ce sort ne fonctionne qu'en extérieur.

\begin{itemize}
\item
  \textbf{Calme :} Élimine les intempéries, même si les effets
  secondaires demeurent.
\item
  \textbf{Chaleur extrême :} Dessèche la neige ou la boue. \\
  Vitesses de déplacement divisées par deux.
\item
  \textbf{Brouillard :} La visibilité tombe à 6 m.\\
  Vitesses de déplacement divisées par deux.\\
  Possibilité de se perdre.
\item
  \textbf{Vents violents :} Projectiles et vol impossibles.\\
  Vitesses de déplacement divisées par deux.\\
  La visibilité tombe à 6 m en zone sablonneuse.\\
  la vitesse de navigation +50\%, si avec le vent.
\item
  \textbf{Pluie :} Malus de --2 aux attaques de projectiles.\\
  Boue en 3 tours~: 50\% vitesse de déplacement.
\item
  \textbf{Neige :} La visibilité tombe à 6 m.\\
  Vitesses de déplacement divisées par deux.\\
  Les plans d'eau peuvent geler et boue après fonte.
\item
  \textbf{Tornade :} Le lanceur peut diriger la tornade haute de 7,20m et large de 1,80m.
  \begin{itemize}
    \item \textbf{CA} 0 {[}19{]}, \textbf{DV} 12* (54 pv), \textbf{Att} 1 × coup
  (2d8), \textbf{TAC0} 10 {[}+9{]}, \textbf{DP} 108 m (36 m) vol,
  \textbf{Sv} MP 6 B 7 PP 8 S 8 SSB 10 (12), \textbf{Ml} 10, \textbf{Al}
  Neutre, \textbf{XP} 1 900, \textbf{NA} 1 (1), \textbf{TT} Aucun
  	\item
  	\textbf{Tourbillon :} Les créatures ayant moins de 2 DV sont balayées
  	(jet de sauvegarde contre la mort).
  	\item
  	\textbf{Immunité aux dégâts normaux :} Ne peut être blessée
  	que par des attaques magiques.
  	\item
  	\textbf{Néfaste aux créatures volantes :} +1d8 de dégâts
  \end{itemize}
\end{itemize}
}
\end{spell}
