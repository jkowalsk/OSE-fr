\begin{spell}
\spellcarac{name}{Paralysie des monstres}
\spellcarac{level}{5}
\spellcarac{duree}{6 tours +1 par niveau}
\spellcarac{portee}{36 m}
\spellcarac{description}{Ce sort paralyse une ou plusieurs créatures qui ratent leur jet de
sauvegarde contre les sorts. Il peut être lancé de deux manières :

\begin{enumerate}
\item
  \textbf{Sur un individu :} Le jet de sauvegarde de la cible se fait
  avec un malus de -- 2.
\item
  \textbf{Sur un groupe :} Le sort affecte 1d4 individus au sein du
  groupe.
\end{enumerate}

\textbf{Restrictions :} Les morts-vivants sont immunisés contre ce sort.
}
\end{spell}
