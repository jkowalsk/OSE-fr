\begin{spell}
\spellcarac{name}{\OE il de magicien}
\spellcarac{level}{4}
\spellcarac{duree}{6 tours}
\spellcarac{portee}{72 m}
\spellcarac{description}{Un \oe il de magicien invisible est invoqué, permettant au lanceur de voir
à distance.

\begin{itemize}
\item
  \textbf{Déplacement :} jusqu'à 36 m par tour.
\item
  \textbf{Voir à travers l'\oe il :} En se concentrant, le lanceur peut
  voir à travers l'\oe il.
\item
  \textbf{Types de vision :} Vision normale et infravision.
\item
  \textbf{Barrières :} Les barrières solides bloquent son déplacement,
  mais il peut passer à travers les trous ou des espaces aussi petits
  que 3 cm de diamètre.
\end{itemize}
}
\end{spell}
