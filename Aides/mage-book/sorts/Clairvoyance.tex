\begin{spell}
\spellcarac{name}{Clairvoyance}
\spellcarac{level}{3}
\spellcarac{duree}{12 tours}
\spellcarac{portee}{18 m}
\spellcarac{description}{Le lanceur gagne la capacité de voir à travers les yeux d'autres
créatures vivantes.

\begin{itemize}
\item Le lanceur doit \textbf{focaliser sa concentration} dans une
  direction \textbf{pendant un tour}.
\item Les \textbf{perceptions visuelles} d'une créature à
  portée dans cette direction sont communiquées au lanceur.
\item Une fois le lien établi, le lanceur peut le
  \textbf{maintenir ou changer de cible}.
\item \textbf{Bloquée} par le plomb ou 60 cm de pierre.
\end{itemize}
}
\end{spell}
