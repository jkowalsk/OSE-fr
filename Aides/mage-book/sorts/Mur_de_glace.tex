\begin{spell}
\spellcarac{name}{Mur de glace}
\spellcarac{level}{4}
\spellcarac{duree}{12 tours}
\spellcarac{portee}{36 m}
\spellcarac{description}{Un mur de glace immobile et translucide jaillit.

\begin{itemize}
\item
  \textbf{Taille :} Toute taille, forme et dimension choisies par le lanceur, jusqu'à 108m$^2$.
\item
  \textbf{Localisation :} Le mur doit reposer sur une surface solide et
  ne peut être invoqué sur des objets présents.
\item
  \textbf{Monstres (3DV ou moins) :} Ils sont incapables de traverser le mur.
\item
  \textbf{Monstres (4 DV ou plus) :} Traverser le mur leur inflige 1d6 points de dégâts.
  Le mur inflige le double de points de dégâts aux créatures utilisant le feu ou y étant accoutumées.
\end{itemize}
}
\end{spell}
