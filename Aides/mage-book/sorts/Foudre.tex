\begin{spell}
\spellcarac{name}{Foudre}
\spellcarac{level}{3}
\spellcarac{duree}{Instantanée}
\spellcarac{portee}{54 m}
\spellcarac{description}{Un puissant éclair d'énergie électrique, de 18 m de long et de 1,50 m de
large, est invoqué.

\begin{itemize}
\item
  \textbf{Dégâts :} 1d6 par niveau du lanceur.
    Sauvegarde contre les sorts divise les dégâts par deux.
\item
  \textbf{Rebond :} Si la foudre frappe une barrière solide, elle est réfléchie et parcourt le
  reste de sa longueur en direction du lanceur.
\end{itemize}
}
\end{spell}
