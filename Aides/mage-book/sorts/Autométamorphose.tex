\begin{spell}
\spellcarac{name}{Autométamorphose}
\spellcarac{level}{4}
\spellcarac{duree}{6 tours +1 par niveau}
\spellcarac{portee}{Le lanceur}
\spellcarac{description}{Le lanceur se métamorphose en une autre créature :

\begin{itemize}
\item
  \textbf{DV :} La nouvelle forme ne doit pas avoir un nombre de Dés de
  vie supérieur au niveau du lanceur.
\item
  \textbf{Statistiques :} Le lanceur conserve son intelligence, ses
  points de vie, ses sauvegardes, ainsi que son TACO.
\item
  \textbf{Capacités physiques :} Celles de la nouvelle forme
  (c'est-à-dire la Force, les formes d'attaques physiques, les modes de
  déplacement, etc.) sont acquises.
\item
  \textbf{Capacités spéciales non physique :} Celles de la nouvelle
  forme (c'est-à-dire les immunités, les souffles, les lancers de sorts)
  ne sont pas acquises.
\item
  \textbf{Lancer de sorts :} Métamorphosé, le lanceur est incapable de
  lancer des sorts.
\item
  \textbf{Individu particulier :} Le sort ne peut pas dupliquer un
  individu particulier.
\item
  \textbf{Retour :} Si le lanceur métamorphosé meurt, il retourne à sa
  forme originelle.
\end{itemize}
}
\end{spell}
