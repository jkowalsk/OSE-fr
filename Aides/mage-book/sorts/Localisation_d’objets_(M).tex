\begin{spell}
\spellcarac{name}{Localisation d'objets}
\spellcarac{level}{2}
\spellcarac{duree}{2 tours}
\spellcarac{portee}{18m + 3 par niveau}
\spellcarac{description}{Le lanceur peut percevoir la direction (mais pas la distance) d'un
objet. Deux types d'objets peuvent être localisés :

\begin{enumerate}
\item
  \textbf{Objet générique :} Un objet générique (par exemple, un
  escalier, un autel, etc.). Dans ce cas, l'objet de ce type le plus
  proche est localisé.
\item
  \textbf{Objet spécifique :} Un objet connu du lanceur.
\item
  \textbf{Restrictions :} Ce sort ne peut pas être utilisé pour localiser
  des créatures.
\end{enumerate}
}
\end{spell}
