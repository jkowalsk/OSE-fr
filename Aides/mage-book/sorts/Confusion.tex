\begin{spell}
\spellcarac{name}{Confusion}
\spellcarac{level}{4}
\spellcarac{duree}{12 rounds}
\spellcarac{portee}{36 m}
\spellcarac{description}{Ce sort rend confuses 3d6 cibles dans un rayon de 18m, incapables de décider de leur action par elles-mêmes.

\begin{itemize}
\item
  \textbf{Cibles de 2+1 DV ou plus :} Sauvegarde contre les sorts chaque round pour résister.
\item
  \textbf{Cibles de 2 DV ou moins :} Sauvegarde impossible.
\item
  \textbf{Comportement :} Pour chaque round et chaque cible, le tableau suivant indique son comportement~:
\end{itemize}

\begin{osrtable}{X[c]X[3]}{1}
2d6 & Comportement \\
2-5 & Attaque le groupe du lanceur \\
6-8 & Aucune action \\
9-12 & Attaque le groupe de la cible \\
\end{osrtable}
}
\end{spell}
