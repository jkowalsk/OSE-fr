\begin{spell}
\spellcarac{name}{Mur de feu}
\spellcarac{level}{4}
\spellcarac{duree}{Concentration}
\spellcarac{portee}{18 m}
\spellcarac{description}{Un rideau immobile et opaque d'un feu violet chatoyant jaillit à portée.

\begin{itemize}
\item
  \textbf{Taille :} Toute taille, forme et dimension choisies par le lanceur, jusqu'à 108 m$^2$.
\item
  \textbf{Localisation :} Ne peut être invoqué sur des objets présents.
\item
\textbf{Monstres (3DV ou moins) :} Ils sont incapables de traverser le mur.
\item
\textbf{Monstres (4 DV ou plus) :} Traverser le mur leur inflige 1d6 points de dégâts.
  Le mur inflige le double de dégâts aux morts-vivants et aux créatures utilisant le froid ou y étant accoutumées.
\item
  \textbf{Concentration :} Le mur perdure tant que le lanceur reste concentré et immobile.
\end{itemize}
}
\end{spell}
