\begin{spell}
\spellcarac{name}{Téléportation}
\spellcarac{level}{5}
\spellcarac{duree}{Instantanée}
\spellcarac{portee}{3m}
\spellcarac{description}{La cible disparaît et réapparaît à un endroit
choisi par le lanceur.

\begin{itemize}
\item
  \textbf{Équipement :} téléporté avec la cible.
\item
  \textbf{Cible réticente :} jet de sauvegarde contre les sorts.
\item
  \textbf{Destination :} La destination doit être :
  \begin{itemize}
  	\item connue du lanceur.
  	\item un espace \textbf{libre} et \textbf{au niveau du sol}.
  \end{itemize}
  Il n'est pas possible de téléporter
  intentionnellement la cible dans les airs ou dans une matière solide.
\end{itemize}

\subsubsection{Résultat de la téléportation}

\resizebox{\columnwidth}{!}{
\begin{osrtable}{llll}{1}
Connaissance & Niveau du sol & Trop haut & Trop bas \\
Faible & 01--50 & 51--75 & 76--00 \\
Moyenne & 01--80 & 81--90 & 91--00 \\
Exacte & 01--95 & 96--99 & 0 \\
\end{osrtable}
}
\begin{itemize}
	\item \textbf{Niveau du sol :} destination souhaitée.
	\item \textbf{Trop haut :} 1d10 x 3 m au-dessus. \\
		Si occupé par de la matière solide, la cible meurt. \\
		Sinon, elle tombe de cette hauteur.
	\item \textbf{Trop bas :} dans le sol et mort instantanée.
	\item \textbf{Connaissance de la destination :}
\begin{itemize}
\item
  \textbf{Faible :} Le lanceur s'y est rendu une ou deux
  fois, il l'a vu par scrutation ou il en a entendu parler.
\item
  \textbf{Moyenne :} Le lanceur s'y est souvent rendu ou
  il l'a étudié plusieurs semaines par scrutation.
\item
  \textbf{Exacte :} Le lanceur a fait une étude détaillée et en personne.
\end{itemize}
\end{itemize}
}
\end{spell}
