\begin{spell}
\spellcarac{name}{Rapidité}
\spellcarac{level}{3}
\spellcarac{duree}{3 tours}
\spellcarac{portee}{72 m}
\spellcarac{description}{Jusqu'à 24 créatures dans une zone de 18 m de diamètre sont enchantées
afin d'être capables de se déplacer et d'agir deux fois plus vite que la
normale :

\begin{itemize}
\item
  \textbf{Déplacement :} Le déplacement (DP) maximum des cibles est
  doublé.
\item
  \textbf{Attaques :} Les cibles peuvent doubler leur nombre d'attaque
  par rounds.
\item
  \textbf{Sorts :} Le nombre de sorts que la cible peut lancer par round
  n'est pas doublé.
\item
  \textbf{Dispositifs magiques :} L'utilisation de dispositifs magiques
  comme les baguettes n'est pas doublée.
\end{itemize}
}
\end{spell}
