\begin{spell}
\spellcarac{name}{Allométamorphose}
\spellcarac{level}{4}
\spellcarac{duree}{Permanente}
\spellcarac{portee}{18 m}
\spellcarac{description}{Une cible vivante se métamorphose :

\begin{itemize}
\item
  \textbf{DV :} Le sort échoue si la nouvelle forme a plus du double de
  DV que la cible.
\item
  \textbf{Points de vie :} La cible conserve son nombre de points de
  vie.
\item
  \textbf{Capacité :} La cible devient réellement la nouvelle forme :
  toutes les capacités spéciales sont acquises, ainsi que le
  comportement, les tendances et l'intelligence.
\item
  \textbf{Individu particulier :} Le sort ne peut pas dupliquer un
  individu particulier.
\item
  \textbf{Retour :} Si la cible meurt, elle retourne à sa forme
  originelle.
\item
  \textbf{Cible non consentante :} Elle peut effectuer un jet de
  sauvegarde contre les sorts pour annuler l'effet.
\end{itemize}
}
\end{spell}
