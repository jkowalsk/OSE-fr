\begin{spell}
\spellcarac{name}{Disque flottant}
\spellcarac{level}{1}
\spellcarac{duree}{6 tours}
\spellcarac{portee}{1.8 m}
\spellcarac{smalldesc}{Disque de 90cm, porte 5 000 pièces (250 kg)}
\spellcarac{description}{Invoque un disque de force circulaire, légèrement concave,
mobile et transportant ce que l'on pose dessus.

\begin{itemize}
\item
  \textbf{Dimensions :} Le disque fait 90 cm de diamètre et 2,50 cm de
  profondeur au centre.
\item
  \textbf{Charge :} Maximum de 5 000 pièces (250 kg).
\item
  \textbf{Déplacement :} Le disque flotte à la hauteur de la taille et
  demeure au même niveau, en se déplaçant horizontalement dans la
  distance indiquée par la portée du sort.
\item
  \textbf{Fin :} Lorsque le disque disparaît à la fin du sort, tout ce
  qu'il transporte tombe au sol.
\end{itemize}
}
\end{spell}
