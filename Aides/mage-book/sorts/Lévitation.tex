\begin{spell}
\spellcarac{name}{Lévitation}
\spellcarac{level}{2}
\spellcarac{duree}{6 tours +1 par niveau}
\spellcarac{portee}{Le lanceur}
\spellcarac{description}{Permet au lanceur de se déplacer dans les airs.

\begin{itemize}
\item
  \textbf{Verticalement :} Le lanceur se dirige mentalement vers le haut
  ou vers le bas et se déplace à une vitesse maximum de 6 m par round.
\item
  \textbf{Horizontalement :} Le lanceur peut s'appuyer sur des objets
  solides pour se déplacer latéralement.
\item
  \textbf{Poids :} La capacité de transport reste normale.
\end{itemize}
}
\end{spell}
