\begin{spell}
\spellcarac{name}{Glissement de terrain}
\spellcarac{level}{6}
\spellcarac{duree}{6 tours}
\spellcarac{portee}{72 m}
\spellcarac{description}{La terre (mais pas la pierre) à portée est réorganisée selon les
souhaits du lanceur.

\begin{itemize}
\item
  \textbf{Déplacement :} Le lanceur peut déplacer la terre dans la zone
  jusqu'à une vitesse de 18 m par tour.
\item
  \textbf{Excavations :} La portée du sort s'étend également vers le
  bas, ce qui permet d'effectuer des excavations.
\end{itemize}
}
\end{spell}
