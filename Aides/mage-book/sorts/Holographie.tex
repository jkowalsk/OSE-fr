\begin{spell}
\spellcarac{name}{Holographie}
\spellcarac{level}{6}
\spellcarac{duree}{6 tours}
\spellcarac{portee}{72 m}
\spellcarac{description}{Une copie illusoire du lanceur apparaît à portée.

\begin{itemize}
\item
  \textbf{Double :} L'image est une duplication exacte du lanceur, que
  l'on ne peut différencier qu'au toucher.
\item
  \textbf{Lancer de sorts ultérieurs :} Ils semblent provenir de
  l'image. Cependant, les cibles doivent toujours être visibles pour le
  lanceur.
\item
  \textbf{Sorts et projectiles :} L'image ne semble pas affectée par les
  sorts ou les projectiles.
\item
  \textbf{Mêlée ou toucher :} Si l'image est touchée ou frappée en
  mêlée, elle disparaît.
\end{itemize}
}
\end{spell}
