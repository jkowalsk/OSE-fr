\begin{spell}
\spellcarac{name}{Détection du mal}
\spellcarac{level}{2}
\spellcarac{duree}{6 tours}
\spellcarac{portee}{18 m}
\spellcarac{description}{Les objets enchantés à des fins maléfiques et les êtres vivants aux
intentions mauvaises émettent une lueur magique.

\begin{itemize}
\item
  \textbf{Intentions seulement :} Ce sort ne permet pas de lire les
  pensées ; il ne donne qu'une impression générale des desseins
  maléfiques d'une personne.
\item
  \textbf{Définition du mal :} En pratique, l'arbitre doit décider de ce
  qui est "maléfique". Notez que des objets potentiellement dangereux,
  comme les pièges, ne sont pas "mauvais".
\end{itemize}
}
\end{spell}
