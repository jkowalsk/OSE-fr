\begin{spell}
\spellcarac{name}{Phytomorphose}
\spellcarac{level}{4}
\spellcarac{duree}{Permanente}
\spellcarac{portee}{72 m}
\spellcarac{description}{Un groupe de créatures de la taille d'un humain, se trouvant dans une
zone de 72 m de diamètre, est cachée par une illusion et prend
l'apparence d'un ensemble d'arbres ou d'un verger.

\begin{itemize}
\item
  \textbf{Traverser :} Une fois l'illusion en place, même les créatures
  traversant les cibles cachées sont trompées.
\item
  \textbf{Cibles quittant la zone :} Cessent d'être
  dissimulées.
\item
  \textbf{Annulation :} Le lanceur peut dissiper l'illusion dans son
  ensemble à tout moment.
\end{itemize}
}
\end{spell}
