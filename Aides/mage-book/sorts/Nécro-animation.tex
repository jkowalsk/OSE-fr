\begin{spell}
\spellcarac{name}{Nécro-animation}
\spellcarac{level}{5}
\spellcarac{duree}{Permanente}
\spellcarac{portee}{18 m}
\spellcarac{description}{Ce sort transforme les os ou les corps de créatures mortes en squelettes
ou zombies morts-vivants :

\begin{itemize}
\item
  \textbf{Obéissants :} Ils obéissent aux ordres du lanceur.
\item
  \textbf{Capacités spéciales :} Incapables d'utiliser les
  capacités spéciales qu'ils possédaient vivants (y compris lancer des sorts).
\item
  \textbf{Durée :} Jusqu'à ce qu'ils soient détruits
  ou qu'un sort dissipation de la magie soit lancé sur eux.
\item
  \textbf{Nombre :} Ce sort anime un nombre de Dés de vie de morts-vivants égal au niveau du lanceur :
\begin{itemize}
\item
  \textbf{Squelettes :} CA 7 {[}12{]} et DV égaux à ceux que la créature
  avait lorsqu'elle était en vie.
\item
  \textbf{Zombies :} CA 8 {[}11{]} et 1 DV en plus par rapport à ceux
  qu'elle avait lorsqu'elle était en vie.
\item
  \textbf{Personnages classés :} Les niveaux de classe ne sont pas
  comptabilisés comme des DV : \\
  un guerrier de 5e niveau réanimé aurait 2 DV (1 DV car humain normal, +1 si réanimé en zombie).
\end{itemize}
\end{itemize}
}
\end{spell}
