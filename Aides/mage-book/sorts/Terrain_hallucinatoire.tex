\begin{spell}
\spellcarac{name}{Terrain hallucinatoire}
\spellcarac{level}{4}
\spellcarac{duree}{Jusqu'à être touché}
\spellcarac{portee}{72 m}
\spellcarac{description}{Terrain hallucinatoire invoque un type de terrain illusoire
  (une colline, une forêt, un marais, etc.), ou dissimule un type de terrain existant.

\begin{itemize}
\item
  \textbf{Zone :} L'illusion doit être entièrement comprise dans la
  portée du sort.
\item
  \textbf{Toucher :} Si l'illusion est touchée par un être intelligent,
  le sort est annulé.
\end{itemize}
}
\end{spell}
