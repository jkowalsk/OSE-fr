\begin{spell}
\spellcarac{name}{Guérison des blessures graves}
\spellcarac{level}{4}
\spellcarac{duree}{Instantanée}
\spellcarac{portee}{Le lanceur ou une créature touchée}
\spellcarac{description}{Le toucher du lanceur restaure 2d6+2 points de vie à une cible vivante.
Les points de vie de la cible ne peuvent pas dépasser son total maximum
habituel.

\subsection{Inversé : Blessure grave}

Inflige 2d6+2 points de dégâts à une créature touchée. En combat, un jet
d'attaque de mêlée est nécessaire.
}
\end{spell}
