\begin{spell}
\spellcarac{name}{Purification de l’eau et des aliments}
\spellcarac{level}{1}
\spellcarac{duree}{Permanent}
\spellcarac{portee}{3 m}
\spellcarac{description}{Ce sort rend la nourriture et l'eau consommables, quel que fût leur état
initial : gâtée, pourrie, empoisonnée, ou contaminée d'une façon
quelconque. Le sort affecte l'un des volumes suivants :

\begin{itemize}
\item
  \textbf{Boisson :} 6 litres de boisson.
\item
  \textbf{Rations :} Une ration de route (standard ou conserve).
\item
  \textbf{Nourritures périssables :} Suffisamment de nourriture
  périssable pour 12 personnes de taille humaine.
\end{itemize}
}
\end{spell}
