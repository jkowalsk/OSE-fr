\begin{spell}
\spellcarac{name}{Aquagenèse}
\spellcarac{level}{4}
\spellcarac{duree}{Permanente}
\spellcarac{portee}{Toucher}
\spellcarac{description}{Ce sort crée une fontaine magique qui surgit du sol ou d'un mur.

\begin{itemize}
\item
  \textbf{Volume :} La fontaine produit environ 220 litres d'eau --
  suffisamment pour assouvir la soif de 12 humains et de 12 montures
  pour une journée.
\item
  \textbf{Lanceurs de sorts de haut niveau :} Si le lanceur est au-delà
  du niveau 8, l'eau assouvit la soif de 12 humains et 12 montures
  supplémentaires pour chaque niveau au-dessus du 8e.
\end{itemize}
}
\end{spell}
