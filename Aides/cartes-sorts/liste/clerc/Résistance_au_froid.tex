\begin{spell}
\spellcarac{name}{Résistance au froid}
\spellcarac{level}{1}
\spellcarac{duree}{6 tours}
\spellcarac{portee}{9 m}
\spellcarac{description}{Toutes les créatures à portée sont protégées contre le froid comme suit
:

\begin{itemize}
\item
  \textbf{Froid normal :} Les créatures ne sont pas affectées par les
  températures froides non magiques.
\item
  \textbf{Bonus de sauvegarde :} Elles reçoivent un bonus de +2 à tous
  les jets de sauvegarde contre les attaques magiques et les souffles
  basés sur le froid.
\item
  \textbf{Dégâts de froid :} Ils sont réduits de 1 point par dé de
  dégâts lancé (chaque dé inflige un minimum de 1 point de dégâts).
\end{itemize}
}
\end{spell}
