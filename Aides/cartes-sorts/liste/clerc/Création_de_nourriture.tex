\begin{spell}
\spellcarac{name}{Création de nourriture}
\spellcarac{level}{5}
\spellcarac{duree}{Permanente}
\spellcarac{portee}{Devant le lanceur}
\spellcarac{description}{Invoque de la nourriture de nulle part.

\begin{itemize}
\item
  \textbf{Volume :} La nourriture est suffisante pour 12 humains et 12
  montures pour une journée.
\item
  \textbf{Lanceurs de sorts de haut niveau :} Si le lanceur est au-delà
  du niveau 8, de la nourriture pour 12 humains et 12 montures
  supplémentaires est invoquée pour chaque niveau au-dessus du 8e.
\end{itemize}
}
