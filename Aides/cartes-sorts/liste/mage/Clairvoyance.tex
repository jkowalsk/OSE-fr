\begin{spell}
\spellcarac{name}{Clairvoyance}
\spellcarac{level}{3}
\spellcarac{duree}{12 tours}
\spellcarac{portee}{18 m}
\spellcarac{description}{Le lanceur gagne la capacité de voir à travers les yeux d'autres
créatures vivantes.

\begin{itemize}
\item
  \textbf{Focalisation pendant 1 tour :} Pour établir un lien avec une
  créature, le lanceur doit focaliser sa concentration dans une
  direction pendant un tour.
\item
  \emph{'Après ce tour :} Les perceptions visuelles d'une créature à
  portée dans cette direction sont communiquées au lanceur.
\item
  \textbf{Permutation :} Une fois le lien établi, le lanceur peut le
  maintenir ou changer de cible.
\item
  \textbf{Obstructions :} La perception des pensées est bloquée par une
  fine couche de plomb ou par une couche de pierre de 60 cm ou plus.
\end{itemize}
}
\end{spell}
