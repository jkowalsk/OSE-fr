\begin{spell}
\spellcarac{name}{Mur de glace}
\spellcarac{level}{4}
\spellcarac{duree}{12 tours}
\spellcarac{portee}{36 m}
\spellcarac{description}{Un mur de glace immobile et translucide jaillit à portée.

\begin{itemize}
\item
  \textbf{Taille :} Le mur peut être de n'importe quelle taille, et de
  toute forme et dimension choisies par le lanceur, jusqu'à 108 m². Par
  exemple, il peut être droit ou recourbé en un cercle protecteur.
\item
  \textbf{Localisation :} Le mur doit reposer sur une surface solide et
  ne peut être invoqué sur des objets présents.
\item
  \textbf{Monstres de moins de 4 DV :} Ils sont incapables de traverser
  le mur de glace.
\item
  \textbf{Monstres de 4 DV ou plus :} Ils peuvent traverser le mur, mais
  subissent 1d6 points de dégâts. Le mur inflige le double de points de
  dégâts aux créatures utilisant le feu ou y étant accoutumées.
\end{itemize}
}
\end{spell}
