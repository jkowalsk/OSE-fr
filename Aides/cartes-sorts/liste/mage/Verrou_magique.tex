\begin{spell}
\spellcarac{name}{Verrou magique}
\spellcarac{level}{2}
\spellcarac{duree}{Permanente}
\spellcarac{portee}{3 m}
\spellcarac{description}{Un verrou magique ferme une porte, un portail, un seuil ou tout objet
muni d'une serrure. Le sort est permanent, mais peut être ignoré comme
suit :

\begin{itemize}
\item
  \textbf{Le lanceur :} Il ignore librement son propre verrou magique.
\item
  \textbf{Sort d'ouverture :} Ce sort permet de passer.
\item
  \textbf{Magiciens de haut niveau :} Un lanceur ayant au moins 3
  niveaux de plus que le lanceur de verrou magique peut passer sans
  encombre.
\item
  \textbf{Temporaire :} Le fait qu'un verrou magique soit ignoré ne met
  pas fin au sort.
\end{itemize}
}
\end{spell}
