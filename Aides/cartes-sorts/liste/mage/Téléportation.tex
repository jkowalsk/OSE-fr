\begin{spell}
\spellcarac{name}{Téléportation}
\spellcarac{level}{5}
\spellcarac{duree}{Instantanée}
\spellcarac{portee}{3 m}
\spellcarac{description}{Le lanceur ou une créature choisie disparaît et réapparaît à un endroit
choisi par le lanceur.

\begin{itemize}
\item
  \textbf{Équipement :} La cible est téléportée avec tout son
  équipement, jusqu'à sa charge maximale.
\item
  \textbf{Cible réticente :} jet de sauvegarde contre les sorts.
\item
  \textbf{Destination :} La destination doit être :
  \begin{itemize}
  	\item connue du lanceur.
  	\item un espace libre au niveau du sol.
  \end{itemize}
  Il n'est pas possible de téléporter
  intentionnellement la cible dans les airs ou dans une matière solide.
\end{itemize}

\subsection{Résultat de la téléportation}

\resizebox{\textwidth}{!}{
\begin{tabular}{llll}
\titlecell Connaissance & \titlecell Niveau du sol & \titlecell Trop haut & \titlecell Trop bas \\
Faible & 01--50 & 51--75 & 76--00 \\
Moyenne & 01--80 & 81--90 & 91--00 \\
Exacte & 01--95 & 96--99 & 0 \\
\end{tabular}
}
\begin{itemize}
	\item \textbf{Niveau du sol :} destination souhaitée.
	\item \textbf{Trop haut :} 1d10 × 3 m au-dessus. \\
		Si occupé par de la matière solide, la cible meurt immédiatement. \\
		Sinon, elle tombe de cette hauteur.
	\item \textbf{Trop bas :} dans le sol et mort instantanée.
	\item \textbf{Connaissance de la destination :}
\begin{itemize}
\item
  \textbf{Faible :} Un endroit où le lanceur s'est rendu une ou deux
  fois, qu'il a vu par scrutation magique ou dont il a entendu parler.
\item
  \textbf{Moyenne :} Un endroit où le lanceur s'est souvent rendu ou
  qu'il a étudié par scrutation pendant plusieurs semaines.
\item
  \textbf{Exacte :} Un endroit dont le lanceur a fait une étude
  détaillée et en personne.
\end{itemize}
\end{itemize}
}
\end{spell}
