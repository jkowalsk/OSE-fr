\begin{spell}
\spellcarac{name}{Invisibilité}
\spellcarac{level}{2}
\spellcarac{duree}{Permanente}
\spellcarac{portee}{72 m}
\spellcarac{description}{Le lanceur, une autre créature, ou un objet à portée devient invisible.

\begin{itemize}
\item
  \textbf{Si lancé sur une créature :} Tout l'équipement que la créature
  porte devient également invisible (incluant son armure et ses
  vêtements). Les objets lâchés ou déposés par une créature invisible
  redeviennent visibles. Si la cible attaque ou lance un sort,
  l'invisibilité est rompue et le sort prend fin.
\item
  \textbf{Si lancé sur un objet :} L'invisibilité est permanente.
\item
  \textbf{Sources de lumière :} Une source lumineuse qui devient
  invisible (qu'elle fasse partie de l'équipement d'une créature ou soit
  un objet individuel) continue à projeter de la lumière.
\end{itemize}
}
\end{spell}
