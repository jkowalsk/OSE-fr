\begin{spell}
\spellcarac{name}{Toile d’araignée}
\spellcarac{level}{2}
\spellcarac{duree}{48 tours}
\spellcarac{portee}{3 m}
\spellcarac{description}{Toile d'araignée invoque une masse enchevêtrée de fibres aussi
résistantes que collantes, qui bloque une zone de 3 m3.

\begin{itemize}
\item
  \textbf{Enchevêtrement :} Les créatures prises dans la toile
  d'araignée se retrouvent empêtrées au milieu de la masse gluante.
  Elles sont incapables de se déplacer, mais peuvent essayer de se
  libérer par la force (voir ci-dessous).
\item
  \textbf{Inflammable :} Les filaments de la toile d'araignée brûlent en
  deux rounds si on y met le feu. Les créatures prises dans la toile en
  feu subissent 1d6 points de dégâts.
\item
  \textbf{Se libérer de la toile :} cela dépend de la Force de la
  créature:

  \begin{itemize}
  \item
    \textbf{Force humaine normale :} La créature se libère en 2d4 tours.
  \item
    \textbf{Force augmentée magiquement au-dessus de 18 :} La créature
    se libère en 4 rounds.
  \item
    \textbf{Force de géant :} La créature se libère en 2 rounds.
  \end{itemize}
\end{itemize}
}
\end{spell}
