\begin{spell}
\spellcarac{name}{Œil de magicien}
\spellcarac{level}{4}
\spellcarac{duree}{6 tours}
\spellcarac{portee}{72 m}
\spellcarac{description}{Un œil de magicien invisible est invoqué, permettant au lanceur de voir
à distance.

\begin{itemize}
\item
  \textbf{Déplacement :} L'œil peut être guidé pour se déplacer à portée
  jusqu'à 36 m par tour.
\item
  \textbf{Voir à travers l'œil :} En se concentrant, le lanceur peut
  voir à travers l'œil.
\item
  \textbf{Types de vision :} L'œil de magicien accorde la vision normale
  et l'infravision.
\item
  \textbf{Barrières :} Les barrières solides bloquent son déplacement,
  mais il peut passer à travers les trous ou des espaces aussi petits
  que 3 cm de diamètre.
\end{itemize}
}
\end{spell}
