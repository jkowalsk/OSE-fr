\begin{spell}
\spellcarac{name}{Désintégration}
\spellcarac{level}{6}
\spellcarac{duree}{Instantanée}
\spellcarac{portee}{18 m}
\spellcarac{description}{La forme matérielle d'une créature ou d'un objet non magique est
instantanément et définitivement détruite.

\begin{itemize}
\item
  \textbf{Si une créature est ciblée :} Elle peut effectuer un jet de
  sauvegarde contre la mort pour résister à la désintégration.
\item
  \textbf{Exemples d'objets :} Les objets suivants peuvent être ciblés :
  un arbre, un navire, une section de 3 m de mur.
\end{itemize}
}
\end{spell}
