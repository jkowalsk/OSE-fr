\begin{spell}
\spellcarac{name}{Paralysie}
\spellcarac{level}{3}
\spellcarac{duree}{9 tours}
\spellcarac{portee}{36 m}
\spellcarac{description}{Ce sort paralyse un ou plusieurs humains, semi-humains ou monstres
humanoïdes qui ratent leur jet de sauvegarde contre les sorts. Il peut
être lancé de deux manières :

\begin{enumerate}
\item
  \textbf{Contre un individu :} Le jet de sauvegarde de la cible se fait
  avec un malus de -- 2.
\item
  \textbf{Sur un groupe :} Le sort affecte 1d4 individus au sein du
  groupe.
\end{enumerate}

\textbf{Restrictions :} Les humanoïdes de plus de 4+1 DV et les
morts-vivants sont immunisés.
}
\end{spell}
