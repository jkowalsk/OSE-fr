\begin{spell}
\spellcarac{name}{Mur de pierre}
\spellcarac{level}{5}
\spellcarac{duree}{Permanente}
\spellcarac{portee}{18 m}
\spellcarac{description}{Un mur de roche solide apparaît.

\begin{itemize}
\item
  \textbf{Taille :} Le mur peut être de la forme souhaitée par le
  lanceur et a un volume de 27 m3 (par exemple, un mur de 60 cm
  d'épaisseur, 15 m de long et 3 m de haut).
\item
  \textbf{Localisation :} Le mur doit reposer sur une surface solide et
  ne peut être invoqué sur des objets présents.
\end{itemize}
}
\end{spell}
