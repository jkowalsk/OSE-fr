\begin{spell}
\spellcarac{name}{Sommeil}
\spellcarac{level}{1}
\spellcarac{duree}{4d4 tours}
\spellcarac{portee}{72 m}
\spellcarac{description}{Ce sort provoque un sommeil magique chez toutes les types de créatures à
l'exception des morts-vivants.

\begin{itemize}
	\item \textbf{Cible:}
\begin{enumerate}
\item
  \textbf{Une créature unique :} Jusqu'à 4+1 DV.
\item
  \textbf{Un groupe :} Un total de 2d8 Dés de vie de créatures ayant 4
  DV ou moins chacune.
\end{enumerate}
\item \textbf{Coup de grâce :} Les créatures endormies par ce sort sont sans
défense et peuvent être tuées instantanément avec une arme tranchante.
\item \textbf{Réveil :} Une blessure ou même une gifle réveille une créature
affectée.
\item \textbf{Si le sort vise un groupe :} Pour les créatures de 4 DV ou
moins, les règles suivantes s'appliquent :
\begin{itemize}
\item
  \textbf{Les faibles d'abord :} Les cibles ayant le moins de DV sont
  affectées en premier.
\item
  \textbf{DV :} Les monstres de moins de 1 DV comptent comme ayant 1 DV,
  tandis que les bonus aux DV sont ignorés (par exemple, un monstre
  ayant 3+2 DV compte comme ayant 3 DV seulement).
\item
  \textbf{Excédent :} Les Dés de vie qui ne sont pas suffisants pour
  affecter une créature sont perdus.
\end{itemize}
\end{itemize}
}
\end{spell}
